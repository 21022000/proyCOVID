--Creacion de la tabla 
CREATE TABLE Vacunacion (
    Nombre VARCHAR(50),
    Apellido VARCHAR(50),
    PuestoLaboral VARCHAR(50),
    VacunaAdministrada VARCHAR(50),
    FechaPrimeraDosis DATE
);

--Creacion del SP
CREATE PROCEDURE DatosVacunacion
AS
BEGIN
    SELECT Nombre, Apellido, PuestoLaboral, VacunaAdministrada, FechaPrimeraDosis
    FROM Vacunacion;
END;
